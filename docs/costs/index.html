ROI Calculation

At what request volume does building your own SLM pay off vs paying the API?

python
# When does building your own SLM pay off?
gpt4_cost_per_1k_tokens = 0.030  # $0.030 per 1K output tokens
avg_tokens_per_request = 200
training_cost = 10  # $10 total
runpod_monthly = 144  # 24/7 RTX 3090

gpt4_monthly_at_n = lambda n: (n * avg_tokens_per_request / 1000) * gpt4_cost_per_1k_tokens
slm_monthly_at_n = lambda n: runpod_monthly  # fixed cost

# Break-even
for n in [1000, 5000, 10000, 50000, 100000]:
    gpt4 = gpt4_monthly_at_n(n)
    slm = slm_monthly_at_n(n)
    print(f'{n:7,} req/mo: GPT-4 ${gpt4:6.0f} | SLM ${slm} | {'SLM wins' if slm < gpt4 else 'GPT-4 wins'}')

# Output:
# GPT-4 wins below ~24,000 requests/month
# SLM wins above ~24,000 requests/month
